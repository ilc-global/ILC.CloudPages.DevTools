// Navigate to a module record
FB.hyperLink('SalesOrder', '50001');  // Opens SO 50001
FB.hyperLink('Customer', '42');       // Opens customer ID 42
FB.hyperLink('Part', 'B201');         // Opens part B201

// Reload the current object in the parent module
FB.reloadObject();

// Status bar
FB.setStatus('Processing 150 records...');
FB.showStatusBar(true);

// Progress bar (0-100, or -1 for indeterminate)
FB.setProgress(50);   // 50%
FB.setProgress(-1);   // Indeterminate/spinning
FB.setProgress(0);    // Reset

// Dialog control
FB.closeDialog();      // Close this CloudPage window
FB.toggleFullscreen(); // Toggle fullscreen mode

// Run a scheduled task
FB.runScheduledTask('Nightly Inventory Sync');
