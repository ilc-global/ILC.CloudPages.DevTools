    // List available printers
var result = FB.localPrinters();
console.log(result.printers);  // ['Printer1', 'Printer2', ...]

// Preview a report (opens Fishbowl preview dialog)
FB.previewReport(42, { SONumber: '50001' });

// Get report as PDF
var result = FB.getReportPDF(42, { SONumber: '50001' });
if (!result.is_error) {
    FB.saveFile('Save Report', 'pdf', 'PDF Files', result.pdf_bytes, 'report.pdf', true);
}

// Print a PDF
FB.printPDF('HP LaserJet', base64PdfData, false);

// Print a report directly
FB.printReportPDF('HP LaserJet', 1, 42, { SONumber: '50001' });

// Print ZPL to label printer
FB.printZPL('Zebra ZT410', '^XA^FO50,50^A0N,50,50^FDHello^FS^XZ');

// Merge and print multiple reports
var dict = [
    { id: 42, params: { SONumber: '50001' } },
    { id: 42, params: { SONumber: '50002' } }
];
FB.printMergedReportsPDF('HP LaserJet', 1, dict);
