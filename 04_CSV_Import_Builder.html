// List all import types
var types = FishbowlCSV.listTypes();

// Get schema and columns
var schema = FishbowlCSV.getSchema('ImportPart');
var columns = FishbowlCSV.getColumns('ImportPart');

// Build import from data
var imp = FishbowlCSV.Part([
    { PartNumber: 'TEST-001', PartDescription: 'Test Part', UOM: 'ea' }
]);
var csv = imp.toCSV();

// Or use the generic create method
var imp = FishbowlCSV.create('ImportPart', data);

// Execute the import
var result = await FB.importCSVAsync('ImportPart', csv);

// Using CsvBuilder for manual CSV
var builder = new FishbowlCSV.CsvBuilder();
builder.header(['PartNumber', 'PartDescription', 'UOM']);
builder.row(['TEST-001', 'Test Part', 'ea']);
var csv = builder.build();

// Import from JSON (alternative)
var result = await FB.importCSVFromJSONAsync('ImportPart',
    JSON.stringify(imp.toRows())
);
