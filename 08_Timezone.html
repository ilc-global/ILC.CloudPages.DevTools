Shows the current time across multiple timezones. Useful for debugging timezone discrepancies.

Timezone Current Time UTC Offset 

// Get current time in server timezone
var serverTime = FB.getTimeForServer('America/Los_Angeles');
// Returns: "2026-03-12 14:30:00"

// Convert server time to client's local time
var clientTime = FB.convertServerTimeToClient(
    '2026-03-12 14:30:00',
    'America/Los_Angeles'
);
// Returns: "2026-03-12 17:30:00" (if client is EST)

// Convert client local time to server timezone
var serverTime = FB.convertClientTimeToServer(
    '2026-03-12 17:30:00',
    'America/Los_Angeles'
);
// Returns: "2026-03-12 14:30:00"

// Common pattern: store server-timezone timestamps
var now = FB.getTimeForServer('America/Los_Angeles');
await FB.queryAsync(
    'UPDATE settings SET lastRun = :time WHERE id = 1',
    { time: now }
);
