// Read a resource file as text
var content = await FB.getResourceFileAsync('sql/my_query.sql');

// Read a resource file as base64
var b64 = await FB.getResourceFileBase64Async('images/logo.png');

// Save a file (native dialog in JXBrowser)
var csvData = btoa('Name,Value\nTest,123');
FB.saveFile('Export Data', 'csv', 'CSV Files', csvData, 'export.csv', true);

// Plugin data — persistent key-value storage per group
// Save data (object or JSON string)
await FB.savePluginDataAsync('my_settings', {
    theme: 'dark',
    pageSize: '50',
    lastRun: '2026-03-12'
});

// Get a specific key
var value = await FB.getPluginDataAsync('my_settings', 'theme');
// Returns: 'dark'

// Delete all data for a group
await FB.deletePluginDataAsync('my_settings');
